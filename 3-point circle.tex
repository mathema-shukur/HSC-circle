	বৃত্তের সাধারণ সমীকরণে $(1,-6)$ বিন্দুটি বসিয়ে পাই  \\ 
	\begin{align*}
			x^2+y^2+2gx+2fy+c&=0\\
			\boxed{\textcolor{blue}{x=1,\,\,y=-6}}&\\
	(1)^2+(-6)^2+2g(1)+2f(-6)+c&=0\\
	\\
	1+36+2g-12f+c&=0\\
	\\
	2g-12f+c&=-37\,\,\,[EQ01]\\
\end{align*}
\\
	বৃত্তের সাধারণ সমীকরণে $(2,1)$ বিন্দুটি বসিয়ে পাই  \\ 
\begin{align*}
		x^2+y^2+2gx+2fy+c&=0\\
		\boxed{\textcolor{blue}{x=2,\,\,y=1}}&\\
	(2)^2+(1)^2+2g(2)+2f(1)+c&=0\\
	\\
	4+1+4g+2f+c&=0\\
	\\
	4g+2f+c&=-5\,\,\,[EQ02]\\
\end{align*}
\\
	বৃত্তের সাধারণ সমীকরণে $(5,2)$ বিন্দুটি বসিয়ে পাই  \\
